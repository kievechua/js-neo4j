  "to": "http://localhost:7474/db/data/node/" + testNode[1]._id,

  "relationships": {
    "type": "testfriend",
    "direction": "out"
  },

  result[0].start.should.equal("http://localhost:7474/db/data/node/" + testNode[0]._id);

});
