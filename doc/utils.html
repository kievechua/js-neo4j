Wrap GET request with promise

get: (url) -> deferred = Q.defer() request .get(url) .type('json') .query(params) .end((err, body) -> if err or not body deferred.reject err else deferred.resolve body ) deferred.promise

Wrap POST request with promise

post: (url, params, success) -> deferred = Q.defer() request .post(url) .type('json') .send(params) .end((err, body) -> if err or not body deferred.reject err else deferred.resolve success(body) ) deferred.promise

Wrap PUT request with promise

put: (url, params) -> deferred = Q.defer() request .put(url) .type('json') .send(params) .end((err, body) -> if err or not body deferred.reject err else deferred.resolve body ) deferred.promise

Wrap Delete request with promise

del: (url) -> deferred = Q.defer() request .del(url) .end((err, body) -> if err deferred.reject err else if body.statusCode >= 400 deferred.reject body.error else deferred.resolve body.ok ) deferred.promise
