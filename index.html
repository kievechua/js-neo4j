Neo4j REST API wrapper for Node.js

Under development Not ready for production, as there's not test case yet.

Github Source

Usage

{Neo4js} = require('neo4js')

neo = new Neo4js()

neo
    .queryBuilder()
    .start('*')
    .return('*')
    .execute()
    .fail((data) ->
        console.log 'error', arguments
    )
    .then((data) ->
        console.log data
    )
